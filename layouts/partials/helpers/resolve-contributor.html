{{/* INPUT: name (string) — contributor display name, e.g. "Gregor Lenz" OUTPUT: dict with keys: Page (Page | nil) — contributor Hugo page if found Name (string) — contributor display name Slug (string) — contributor slug ImageResource (Resource | nil) — page-bundle image resource if found */}} {{ $name := . }} {{ $page := "" }} {{/* 1. Try to find the exact contributor page by matching the Title front matter */}} {{/* This is 100% robust against special characters (é, ć, ț, etc.) */}} {{ $contributors := where site.RegularPages "Type" "contributors" }} {{ range $contributors }} {{ if eq (lower .Title) (lower $name) }} {{ $page = . }} {{ end }} {{ end }} {{/* 2. Fallback to slug-based path lookup if no title matched */}} {{ if not $page }} {{ $slug_guess := $name | urlize }} {{ $page = site.GetPage (printf "contributors/%s" $slug_guess) }} {{ end }} {{/* 3. Determine the final slug to return */}} {{ $slug := "" }} {{ if $page }} {{ $slug = path.Base $page.File.Dir }} {{ else }} {{ $slug = $name | urlize }} {{ end }} {{/* 4. Resolve the image resource from the page bundle */}} {{ $image := "" }} {{ if and $page $page.Params.image }} {{ with $page.Resources.GetMatch $page.Params.image }}{{ $image = . }}{{ end }} {{ end }} {{ return (dict "Page" $page "Name" $name "Slug" $slug "ImageResource" $image) }}
